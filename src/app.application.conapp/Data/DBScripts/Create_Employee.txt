CREATE TABLE [dbo].[Employees] (
    [Id]                 INT           NOT NULL PRIMARY KEY,
    [FirstName]          VARCHAR (32)  NOT NULL,
    [LastName]           VARCHAR (32)  NOT NULL,
    [JobRoleId]          INT           NULL,
    [OriginalHireDate]   DATE          NULL,
    [LocationId]         INT           NULL,
    [DeptId]             INT           NULL,
    [ContractId]         INT           NULL,
    [ManagerId]          INT           NULL,
    [DivisionId]         INT           NULL,
    [Email]              VARCHAR (128) NULL,
    [TerminationDate]    DATE          NULL,
    [CompensationPlanId] INT           NULL,
	FOREIGN KEY (JobRoleId) REFERENCES JobRoles(Id),
	FOREIGN KEY (LocationId) REFERENCES Locations(Id),
	FOREIGN KEY (DeptId) REFERENCES Departments(Id),
	FOREIGN KEY (ContractId) REFERENCES Contracts(Id),
	FOREIGN KEY (ManagerId) REFERENCES Employees(Id),
	FOREIGN KEY (DivisionId) REFERENCES Divisions(Id),
	FOREIGN KEY (CompensationPlanId) REFERENCES CompensationPlans(Id)
);
